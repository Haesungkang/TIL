#### Atoi

````python
def atoi(str):
    value = 0
    for i in range(len(str)):
        c = str[i]
        # 0 ~ 9
        # if c >= '0' and c <= '9':
        # if '0' <= c <= '9':
        #     digit = ord(c) - ord('0')  # c - '0' (x)
        # else:
        #     break
        value = value * 10 + ord(c) - ord('0')
    return value

a = "123"
print(type(a))
b = atoi(a)
print(b, type(b))
````

def itoa(num):
    x = num
    y = 0 # 나머지
    arr = []
    while x:
        y = x % 10
        x = x // 10 # x //=10
        arr.append(chr(y + ord('0')))

    arr.reverse()
    # str = "".join(arr)
    return arr

#### Bruteforce

```python
'''
i(target) j(pattern)

'''

p = "is"
t = "This is a book!"
M = len(p)
N = len(t)

def bruteForce(p, t):
    i = 0
    j = 0
    while j < M and i < N :
        if t[i] != p[j]: # 매칭에 실패하면
            i = i -j
            j = -1 # 24번행에서 +1하면 0으로 변경되므로 -> 첫번째 패턴문자로 이동
        i += 1
        j += 1
    if j == M : return i - M
    else: return -1
```

def str_reve(str):
    arr = list(str)
    for i in range(len(arr)//2):
        arr[i], arr[len(arr)-1-i] = arr[len(arr)-1-i], arr[i]
        str = ''.join(arr)

def strcmp(s1, s2):
    i = 0
    if len(s1) != len(s2):
        return False
    else:
        i = 0 # 초기식
        while i < len(s1) and i < len(s2): # 조건식
            if s1[i] != s2[i]:
                return False
            i += 1 #증감식
    return True
